Inicializacao
- flag_spawn <- 1 //flag_spawn, caso 1, spawna peca --- PRONTO

main
- imprime mensagem inicial --- PRONTO
- wait_start --- PRONTO
	- var count
		-> inc count -> reset a cada 499
			-> condicao de stop : enter
	- wait enter
- inicia jogo --- PRONTO
	-> desenha_mapa
	-> index inicial do vetor randomico <- count
	-> t_peca <- rand[index inicial]
- spawn peca
	-> verifica flag_spawn //caso 0, skip spawn peca
	-> verificar t_peca
	-> verifica se é possível spawn //caso não seja, GAME OVER
	-> definir posição de spawn
	-> desenha peca
	-> flag_spawn <- 0

MAIN LOOP
	- mover peca
		-> verificar input e recalcula posicao
			-> mov esquerda
			-> mov direita
			-> mov baixo (automático)
			-> rotaciona
			-> mov baixo pelo input *
			-> descer de uma vez *
			-> apaga desenho da peca
			-> desenha peca
	- verifica colisao
		-> setar flags de colisão
			-> colisao em baixo
			-> colisao esquerda
			-> colisao direita
			-> colisao em cima *
		-> se colisao em baixo, entao
			-> fixa peca
				-> atualiza mapa
				-> flag_spawn <- 1
				-> vetor temporario da ultima peca <- posicao dos quadradinhos da peca fixada (para depois dividir por 40 (divisao inteira))
	- verifica se completa linha
		-> se True move tudo que esta acima para baixo
			-> pontuacao++
	- spawn peca
